1138.字母板上的路径（中等）

[1138.字母板上的路径（中等）](https://leetcode.cn/problems/alphabet-board-path/)

解法：模拟

对上下走分情况讨论可以避开z的坑

```go
func alphabetBoardPath(target string) string {
	var buffer bytes.Buffer
	prei, prej := 0, 0
	for _, c := range target {
		ci, cj := int((c-'a')/5), int((c-'a')%5)
		if dir := prei < ci; dir {
			//向下移动，先左右再下
			if prej < cj {
				buffer.Write([]byte(strings.Repeat("R", cj-prej)))
			} else {
				buffer.Write([]byte(strings.Repeat("L", prej-cj)))
			}
			buffer.Write([]byte(strings.Repeat("D", ci-prei)))
		} else {
			//向上移动，先上再左右
			buffer.Write([]byte(strings.Repeat("U", prei-ci)))
			if prej < cj {
				buffer.Write([]byte(strings.Repeat("R", cj-prej)))
			} else {
				buffer.Write([]byte(strings.Repeat("L", prej-cj)))
			}
		}
		buffer.WriteRune('!')
		prei, prej = ci, cj
	}

	return buffer.String()
}
```
