1014.最佳观光组合
[1014.最佳观光组合](https://leetcode-cn.com/problems/best-sightseeing-pair/)

解法：贪心
```go
func maxScoreSightseeingPair(values []int) int {
	result:=0
	preMax:=values[0]
	for i:=1;i<len(values);i++ {
		now:=values[i]
		preMax--
		if now+preMax>result {
			result=now+preMax
		}
		if now>preMax {
			preMax=now
		}
	}
	return result
}
